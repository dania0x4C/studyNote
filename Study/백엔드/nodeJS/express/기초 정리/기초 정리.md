```
var express = require('express');// 모듈가지고 오기
var app = express(); // 할당

// 라우팅 설정부분
// 첫 번째 인자: 요청 url 경로
// 두 번째 인자: 요청 function
// request: 요청인자
// response: 응답인자 
app.get('/', function (req, res) {
	res.send('Hello World!');
});

app.listen(3030, function () {
	console.log('Example app listening on port 3030!');
});
```
